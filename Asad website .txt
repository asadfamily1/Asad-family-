  <script>
    // Simple bilingual toggle (Oromo <-> English). You can expand translations as needed.
    (function(){
      const translations = {
        om: {
          title: "Baga Gara Weebsayitii Keenyaa Nagaan Dhuftan!",
          subtitle: "Tajaajila keenya beeksisuuf qophiidha.",
          learn: "Odeeffannoo Dabalataa",
          contact: "Nu Quunnamaa",
          langLabel: "English"
        },
        en: {
          title: "Welcome to Our Website!",
          subtitle: "We are ready to showcase our services.",
          learn: "Learn more",
          contact: "Contact",
          langLabel: "Afaan Oromoo"
        }
      };

      let current = 'om';
      const titleEl = document.getElementById('page-title');
      const subtitleEl = document.getElementById('subtitle');
      const learnBtn = document.getElementById('learnBtn');
      const contactBtn = document.getElementById('contactBtn');
      const langBtn = document.getElementById('langBtn');

      function applyLang(lang){
        const t = translations[lang];
        document.documentElement.lang = lang;
        titleEl.textContent = t.title;
        titleEl.lang = lang;
        subtitleEl.textContent = t.subtitle;
        subtitleEl.lang = lang;
        learnBtn.firstChild && learnBtn.removeChild(learnBtn.firstChild);
        learnBtn.textContent = t.learn + " / " + (lang === 'om' ? 'Learn more' : 'Odeeffannoo Dabalataa');
        contactBtn.textContent = t.contact + " / " + (lang === 'om' ? 'Contact' : 'Nu Quunnamaa');
        langBtn.textContent = t.langLabel;
      }

      langBtn.addEventListener('click', () => {
        current = (current === 'om') ? 'en' : 'om';
        applyLang(current);
      });

      // initial
      applyLang(current);
    })();
  </script>
